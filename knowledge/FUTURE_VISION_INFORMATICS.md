            return self.synthesize_waves(wave_signature)
        else:
            raise InsufficientUnderstanding("더 공부해야 합니다")
    
    def can_reproduce_anything(self):
        """모든 것을 파동으로 재현할 수 있는가?"""
        return self.understanding == COMPLETE_WAVE_UNDERSTANDING
